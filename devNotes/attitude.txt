function drawAttitudeNeedle(pitchDeg, bankDeg) {
    const canvas = document.getElementById("attCanvas");
    const ctx    = canvas.getContext("2d");
    const w      = canvas.width;
    const h      = canvas.height;
    const cx     = w / 2;
    const cy     = h / 2;

    ctx.clearRect(0, 0, w, h);

    ctx.save();
    ctx.translate(cx, cy);

    // Roll: rotate airplane symbol with bank
    const bankRad = bankDeg * Math.PI / 180;
    ctx.rotate(bankRad);

    // Pitch: move the “wings” up/down slightly
    const pitchScale = 2; // pixels per degree
    const pitchOffset = -pitchDeg * pitchScale;

    ctx.translate(0, pitchOffset);

    // === Draw central airplane symbol ===
    ctx.lineWidth   = 4;
    ctx.strokeStyle = "#ffffff";

    const wingSpan = 80;
    const fuselage = 40;

    // Wings
    ctx.beginPath();
    ctx.moveTo(-wingSpan / 2, 0);
    ctx.lineTo( wingSpan / 2, 0);
    ctx.stroke();

    // Fuselage (vertical bar)
    ctx.beginPath();
    ctx.moveTo(0, 0);
    ctx.lineTo(0, fuselage);
    ctx.stroke();

    // Small center dot
    ctx.beginPath();
    ctx.arc(0, 0, 4, 0, Math.PI * 2);
    ctx.fillStyle = "#ffffff";
    ctx.fill();

    ctx.restore();
}


function updateAttitude() {
    // These should come from your SimConnect / stub
    const pitch = gsdPitchDeg;  // e.g. PLANE_PITCH_DEGREES
    const bank  = gsdBankDeg;   // e.g. PLANE_BANK_DEGREES

    drawAttitudeNeedle(pitch, bank);
}
