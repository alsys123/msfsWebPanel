def simconnect_set(var, value):
    global sm, aq

    if not simconnect_available:
        return False

    try:
        if sm is None:
            sm = SimConnect()
            aq = AircraftRequests(sm, _time=0)

        aq.set(var, value)
        return True

    except Exception as e:
        print("SimConnect SET error:", e)
        sm = None
        return False
